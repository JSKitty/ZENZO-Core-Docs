127 static int secp256k1_ecmult_wnaf(int *wnaf, const secp256k1_scalar_t *a, int w) {
128  secp256k1_scalar_t s = *a;
129 
130  int sign = 1;
131  if (secp256k1_scalar_get_bits(&s, 255, 1)) {
132  secp256k1_scalar_negate(&s, &s);
133  sign = -1;
134  }
135 
136  int set_bits = 0;
137  int bit = 0;
138  while (bit < 256) {
139  if (secp256k1_scalar_get_bits(&s, bit, 1) == 0) {
140  bit++;
141  continue;
142  }
143  while (set_bits < bit) {
144  wnaf[set_bits++] = 0;
145  }
146  int now = w;
147  if (bit + now > 256) {
148  now = 256 - bit;
149  }
150  int word = secp256k1_scalar_get_bits_var(&s, bit, now);
151  if (word & (1 << (w-1))) {
152  secp256k1_scalar_add_bit(&s, bit + w);
153  wnaf[set_bits++] = sign * (word - (1 << w));
154  } else {
155  wnaf[set_bits++] = sign * word;
156  }
157  bit += now;
158  }
159  return set_bits;
160 }

162 static void secp256k1_ecmult(secp256k1_gej_t *r, const secp256k1_gej_t *a, const secp256k1_scalar_t *na, const secp256k1_scalar_t *ng) {

164 
165 #ifdef USE_ENDOMORPHISM
166  secp256k1_scalar_t na_1, na_lam;
167  /* split na into na_1 and na_lam (where na = na_1 + na_lam*lambda, and na_1 and na_lam are ~128 bit) */
168  secp256k1_scalar_split_lambda_var(&na_1, &na_lam, na);
169 
170  /* build wnaf representation for na_1 and na_lam. */

173  VERIFY_CHECK(bits_na_1 <= 130);
174  VERIFY_CHECK(bits_na_lam <= 130);
175  int bits = bits_na_1;
176  if (bits_na_lam > bits) bits = bits_na_lam;
177 #else
178  /* build wnaf representation for na. */

181 #endif
182 
183  /* calculate odd multiples of a */
184  secp256k1_gej_t pre_a[ECMULT_TABLE_SIZE(WINDOW_A)];
185  secp256k1_ecmult_table_precomp_gej_var(pre_a, a, WINDOW_A);
186 
187 #ifdef USE_ENDOMORPHISM

190  secp256k1_gej_mul_lambda(&pre_a_lam[i], &pre_a[i]);

201  if (bits_ng_1 > bits) bits = bits_ng_1;
202  if (bits_ng_128 > bits) bits = bits_ng_128;
203 #else
204  int wnaf_ng[257]; int bits_ng = secp256k1_ecmult_wnaf(wnaf_ng, ng, WINDOW_G);
205  if (bits_ng > bits) bits = bits_ng;
206 #endif
207 
208  secp256k1_gej_set_infinity(r);
209  secp256k1_gej_t tmpj;
210  secp256k1_ge_t tmpa;
211 
212  for (int i=bits-1; i>=0; i--) {
213  secp256k1_gej_double_var(r, r);
214  int n;
215 #ifdef USE_ENDOMORPHISM
216  if (i < bits_na_1 && (n = wnaf_na_1[i])) {
217  ECMULT_TABLE_GET_GEJ(&tmpj, pre_a, n, WINDOW_A);
218  secp256k1_gej_add_var(r, r, &tmpj);
219  }

221  ECMULT_TABLE_GET_GEJ(&tmpj, pre_a_lam, n, WINDOW_A);
222  secp256k1_gej_add_var(r, r, &tmpj);
223  }
224  if (i < bits_ng_1 && (n = wnaf_ng_1[i])) {
225  ECMULT_TABLE_GET_GE(&tmpa, c->pre_g, n, WINDOW_G);
226  secp256k1_gej_add_ge_var(r, r, &tmpa);
227  }
228  if (i < bits_ng_128 && (n = wnaf_ng_128[i])) {
229  ECMULT_TABLE_GET_GE(&tmpa, c->pre_g_128, n, WINDOW_G);
230  secp256k1_gej_add_ge_var(r, r, &tmpa);
231  }
232 #else
233  if (i < bits_na && (n = wnaf_na[i])) {
234  ECMULT_TABLE_GET_GEJ(&tmpj, pre_a, n, WINDOW_A);
235  secp256k1_gej_add_var(r, r, &tmpj);

239  secp256k1_gej_add_ge_var(r, r, &tmpa);

242  }
