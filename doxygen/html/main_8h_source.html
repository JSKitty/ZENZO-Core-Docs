542 class CValidationState

544 private:

546  MODE_VALID,
547  MODE_INVALID,
548  MODE_ERROR,
549  } mode;
550  int nDoS;
551  std::string strRejectReason;
552  unsigned char chRejectCode;
553  bool corruptionPossible;
554 
555 public:
556  CValidationState() : mode(MODE_VALID), nDoS(0), chRejectCode(0), corruptionPossible(false) {}
557  bool DoS(int level, bool ret = false, unsigned char chRejectCodeIn = 0, std::string strRejectReasonIn = "", bool corruptionIn = false)
558  {
559  chRejectCode = chRejectCodeIn;
560  strRejectReason = strRejectReasonIn;
561  corruptionPossible = corruptionIn;
562  if (mode == MODE_ERROR)
563  return ret;
564  nDoS += level;
565  mode = MODE_INVALID;
566  return ret;
567  }
568  bool Invalid(bool ret = false,
569  unsigned char _chRejectCode = 0,
570  std::string _strRejectReason = "")
571  {
572  return DoS(0, ret, _chRejectCode, _strRejectReason);
573  }
574  bool Error(std::string strRejectReasonIn = "")
575  {
576  if (mode == MODE_VALID)
577  strRejectReason = strRejectReasonIn;
578  mode = MODE_ERROR;
579  return false;
580  }
581  bool Abort(const std::string& msg)
582  {
583  AbortNode(msg);
584  return Error(msg);
585  }
586  bool IsValid() const
587  {
588  return mode == MODE_VALID;
589  }
590  bool IsInvalid() const
591  {
592  return mode == MODE_INVALID;
593  }
594  bool IsError() const
595  {
596  return mode == MODE_ERROR;
597  }
598  bool IsInvalid(int& nDoSOut) const
599  {
600  if (IsInvalid()) {
601  nDoSOut = nDoS;
602  return true;
603  }
604  return false;
605  }
606  bool CorruptionPossible() const
607  {
608  return corruptionPossible;
609  }
610  unsigned char GetRejectCode() const { return chRejectCode; }
611  std::string GetRejectReason() const { return strRejectReason; }
