175  unsigned int GetSerializeSize(int nType, int nVersion) const
176  {
177  unsigned int nSize = 0;
178  unsigned int nMaskSize = 0, nMaskCode = 0;
179  CalcMaskSize(nMaskSize, nMaskCode);
180  bool fFirst = vout.size() > 0 && !vout[0].IsNull();
181  bool fSecond = vout.size() > 1 && !vout[1].IsNull();
182  assert(fFirst || fSecond || nMaskCode);
183  unsigned int nCode = 8 * (nMaskCode - (fFirst || fSecond ? 0 : 1)) + (fCoinBase ? 1 : 0) + (fCoinStake ? 2 : 0) + (fFirst ? 4 : 0) + (fSecond ? 8 : 0);
184  // version
185  nSize += ::GetSerializeSize(VARINT(this->nVersion), nType, nVersion);
186  // size of header code
187  nSize += ::GetSerializeSize(VARINT(nCode), nType, nVersion);
188  // spentness bitmask
189  nSize += nMaskSize;
190  // txouts themself
191  for (unsigned int i = 0; i < vout.size(); i++)
192  if (!vout[i].IsNull())
193  nSize += ::GetSerializeSize(CTxOutCompressor(REF(vout[i])), nType, nVersion);
194  // height
195  nSize += ::GetSerializeSize(VARINT(nHeight), nType, nVersion);
196  return nSize;
197  }

199  template <typename Stream>
200  void Serialize(Stream& s, int nType, int nVersion) const
201  {
202  unsigned int nMaskSize = 0, nMaskCode = 0;
203  CalcMaskSize(nMaskSize, nMaskCode);
204  bool fFirst = vout.size() > 0 && !vout[0].IsNull();
205  bool fSecond = vout.size() > 1 && !vout[1].IsNull();
206  assert(fFirst || fSecond || nMaskCode);
207  unsigned int nCode = 16 * (nMaskCode - (fFirst || fSecond ? 0 : 1)) + (fCoinBase ? 1 : 0) + (fCoinStake ? 2 : 0) + (fFirst ? 4 : 0) + (fSecond ? 8 : 0);
208  // version
209  ::Serialize(s, VARINT(this->nVersion), nType, nVersion);
210  // header code
211  ::Serialize(s, VARINT(nCode), nType, nVersion);
212  // spentness bitmask
213  for (unsigned int b = 0; b < nMaskSize; b++) {
214  unsigned char chAvail = 0;
215  for (unsigned int i = 0; i < 8 && 2 + b * 8 + i < vout.size(); i++)
216  if (!vout[2 + b * 8 + i].IsNull())
217  chAvail |= (1 << i);
218  ::Serialize(s, chAvail, nType, nVersion);
219  }
220  // txouts themself
221  for (unsigned int i = 0; i < vout.size(); i++) {
222  if (!vout[i].IsNull())
223  ::Serialize(s, CTxOutCompressor(REF(vout[i])), nType, nVersion);
224  }
225  // coinbase height
226  ::Serialize(s, VARINT(nHeight), nType, nVersion);
227  }

229  template <typename Stream>
230  void Unserialize(Stream& s, int nType, int nVersion)
231  {
232  unsigned int nCode = 0;
233  // version
234  ::Unserialize(s, VARINT(this->nVersion), nType, nVersion);
235  // header code
236  ::Unserialize(s, VARINT(nCode), nType, nVersion);
237  fCoinBase = nCode & 1; //0001 - means coinbase
238  fCoinStake = (nCode & 2) != 0; //0010 coinstake
239  std::vector<bool> vAvail(2, false);
240  vAvail[0] = (nCode & 4) != 0; // 0100
241  vAvail[1] = (nCode & 8) != 0; // 1000
242  unsigned int nMaskCode = (nCode / 16) + ((nCode & 12) != 0 ? 0 : 1);
243  // spentness bitmask
244  while (nMaskCode > 0) {
245  unsigned char chAvail = 0;
246  ::Unserialize(s, chAvail, nType, nVersion);
247  for (unsigned int p = 0; p < 8; p++) {
248  bool f = (chAvail & (1 << p)) != 0;
249  vAvail.push_back(f);
250  }
251  if (chAvail != 0)
252  nMaskCode--;
253  }
254  // txouts themself
255  vout.assign(vAvail.size(), CTxOut());
256  for (unsigned int i = 0; i < vAvail.size(); i++) {
257  if (vAvail[i])
258  ::Unserialize(s, REF(CTxOutCompressor(vout[i])), nType, nVersion);
259  }
260  // coinbase height
261  ::Unserialize(s, VARINT(nHeight), nType, nVersion);
262  Cleanup();
263  }
