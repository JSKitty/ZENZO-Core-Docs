313 inline unsigned int GetSizeOfCompactSize(uint64_t nSize)
314 {
315  if (nSize < 253)
316  return sizeof(unsigned char);
317  else if (nSize <= std::numeric_limits<unsigned short>::max())
318  return sizeof(unsigned char) + sizeof(unsigned short);
319  else if (nSize <= std::numeric_limits<unsigned int>::max())
320  return sizeof(unsigned char) + sizeof(unsigned int);
321  else
322  return sizeof(unsigned char) + sizeof(uint64_t);
323 }

325 template <typename Stream>
326 void WriteCompactSize(Stream& os, uint64_t nSize)
327 {
328  if (nSize < 253) {
329  unsigned char chSize = nSize;
330  WRITEDATA(os, chSize);
331  } else if (nSize <= std::numeric_limits<unsigned short>::max()) {
332  unsigned char chSize = 253;
333  unsigned short xSize = nSize;
334  WRITEDATA(os, chSize);
335  WRITEDATA(os, xSize);
336  } else if (nSize <= std::numeric_limits<unsigned int>::max()) {
337  unsigned char chSize = 254;
338  unsigned int xSize = nSize;
339  WRITEDATA(os, chSize);
340  WRITEDATA(os, xSize);
341  } else {
342  unsigned char chSize = 255;
343  uint64_t xSize = nSize;
344  WRITEDATA(os, chSize);
345  WRITEDATA(os, xSize);
346  }
347  return;
348 }

350 template <typename Stream>
351 uint64_t ReadCompactSize(Stream& is)
352 {
353  unsigned char chSize;
354  READDATA(is, chSize);
355  uint64_t nSizeRet = 0;
356  if (chSize < 253) {
357  nSizeRet = chSize;
358  } else if (chSize == 253) {
359  unsigned short xSize;
360  READDATA(is, xSize);
361  nSizeRet = xSize;
362  if (nSizeRet < 253)
363  throw std::ios_base::failure("non-canonical ReadCompactSize()");
364  } else if (chSize == 254) {
365  unsigned int xSize;
366  READDATA(is, xSize);
367  nSizeRet = xSize;
368  if (nSizeRet < 0x10000u)
369  throw std::ios_base::failure("non-canonical ReadCompactSize()");
370  } else {
371  uint64_t xSize;
372  READDATA(is, xSize);
373  nSizeRet = xSize;
374  if (nSizeRet < 0x100000000ULL)
375  throw std::ios_base::failure("non-canonical ReadCompactSize()");
376  }
377  if (nSizeRet > (uint64_t)MAX_SIZE)
378  throw std::ios_base::failure("ReadCompactSize() : size too large");
379  return nSizeRet;
380 }

406 template <typename I>
407 inline unsigned int GetSizeOfVarInt(I n)
408 {
409  int nRet = 0;
410  while (true) {
411  nRet++;
412  if (n <= 0x7F)
413  break;
414  n = (n >> 7) - 1;
415  }
416  return nRet;
417 }

419 template <typename Stream, typename I>
420 void WriteVarInt(Stream& os, I n)
421 {
422  unsigned char tmp[(sizeof(n) * 8 + 6) / 7];
423  int len = 0;
424  while (true) {
425  tmp[len] = (n & 0x7F) | (len ? 0x80 : 0x00);
426  if (n <= 0x7F)
427  break;
428  n = (n >> 7) - 1;
429  len++;
430  }
431  do {
432  WRITEDATA(os, tmp[len]);
433  } while (len--);
434 }

436 template <typename Stream, typename I>
437 I ReadVarInt(Stream& is)
438 {
439  I n = 0;
440  while (true) {
441  unsigned char chData;
442  READDATA(is, chData);
443  n = (n << 7) | (chData & 0x7F);
444  if (chData & 0x80)
445  n++;
446  else
447  return n;
448  }
449 }
